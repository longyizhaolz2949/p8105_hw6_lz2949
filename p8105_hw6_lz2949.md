``` r
# comparison in terms of cross-validated prediction error; use corssv_mc and functions in purrr. 
b_df =
  crossv_mc(bwt_df, 100) |> 
  mutate(
    train = map(train, as_tibble),
    test = map(test, as_tibble))

# use mutate +map and map 2 to fit models
b_df = 
  b_df |> 
  mutate(
    prop_lm  = map(train, \(df) lm(bwt ~ wtgain, data = df)),
    lbga_mod = map(train, \(df) lm(bwt ~ blength + gaweeks, data = df)),
    hcls_mod = map(train, \(df) lm(bwt ~ bhead + blength + babysex + bhead * blength * babysex, data = df))) |> 
  mutate(
    prop_linear = map2_dbl(prop_lm, test, \(mod, df) rmse(model = mod, data = df)),
    lbga_linear = map2_dbl(lbga_mod, test, \(mod, df) rmse(model = mod, data = df)),
    hcls_smooth = map2_dbl(hcls_mod, test, \(mod, df) rmse(model = mod, data = df)))
```
